  %% Document Class and Margins %%

	\usepackage[table]{xcolor}	% Must be above tikz or graphicx for precedence

	\usepackage{graphicx}

	% Package for underline and strikethrough fonts %
	\usepackage[normalem]{ulem}
	
	% Package for including PDF documents %
	\usepackage{pdfpages}
	
	% Package for commenting large sections of text %
	\usepackage{comment}
	
	% Package for full features outlines %
	\usepackage{outlines}

	% Package for creating long, cross page tables %
	\usepackage{longtable}

	% Package for spanning multiple rows in tables %
	\usepackage{multirow}

	% Package for turning schedule to landscape %
	\usepackage{lscape}
	
	% Package for Biolinum/Libertine Font %
	\usepackage[T1]{fontenc}
	\usepackage{libertine}

	\usepackage[explicit,pagestyles]{titlesec}

	\usepackage{fancyhdr}

	\renewcommand*\familydefault{\sfdefault}

	\renewcommand{\footrulewidth}{0pt}

	\lhead{} %blank in all right header spaces
	\chead{} %blank in all right header spaces

	\rfoot{{\bfseries My Company Name}}

	\newcommand{\HRule}{\rule{\linewidth}{0.5mm}}

	% Left align longtable environments
	\setlength{\LTleft}{0pt}

	\titleformat{name=\section}

	{0pt}

	\titleformat{name=\subsubsection}
	{\normalfont\bfseries}
	{}{0pt}{\color{Color1}\hspace{2em} #1}
